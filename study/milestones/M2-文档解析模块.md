export async function parseDocument(file: File, filename: string): Promise<string> {
  const ext = filename.split('.').pop()?.toLowerCase();
  
  switch (ext) {
    case 'pdf':
      return await parsePDF(file);
    case 'docx':
      return await parseDOCX(file);
    case 'xlsx':
      return await parseXLSX(file);
    case 'txt':
    case 'md':
      return await file.text();
    default:
      throw new Error(`不支持的文件格式: ${ext}`);

}

export function chunkTextWithMetadata(text: string, filename: string): TextChunk[] {
  const chunks: TextChunk[] = [];
  
  // 1. 按标题分段
  const sections = splitByHeadings(text);
  
  // 2. 对每个段落进行固定大小切片
  sections.forEach((section, index) => {
    if (section.text.length <= 1000) {
      chunks.push({
        text: section.text,
        metadata: {
          filename,
          heading: section.heading,
          chunkIndex: chunks.length
        }
      });
    } else {
      // 大段落切片（带重叠）
      const subChunks = splitWithOverlap(section.text, 1000, 100);
      subChunks.forEach(text => {
        chunks.push({
          text,
          metadata: {
            filename,
            heading: section.heading,
            chunkIndex: chunks.length
          }
        });
      });
    }
  });
  
  return chunks;
}

## ⚠️ 注意事项

1. **PDF 解析需要 Canvas 环境**：Node.js 环境需配置 canvas polyfill
2. **XLSX 解析性能**：大文件可能耗时较长，考虑流式处理
3. **标题提取正确性**：测试多级标题、非标准格式
4. **切片重叠的意义**：保持上下文连续性，避免语义截断
5. **元数据完整性**：后续向量搜索需要这些元数据追溯来源
