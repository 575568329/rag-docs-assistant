    // 计算余弦相似度
    function cosineSimilarity(a: number[], b: number[]): number {
      const dotProduct = a.reduce((sum, val, i) => sum + val * b[i], 0);
      const normA = Math.sqrt(a.reduce((sum, val) => sum + val * val, 0));
      const normB = Math.sqrt(b.reduce((sum, val) => sum + val * val, 0));
      return dotProduct / (normA * normB);
    }

    // Reciprocal Rank Fusion
    function rrf(rankings: Array<{idx: number, rank: number}[]>, k = 60): Map<number, number> {
      const scores = new Map<number, number>();
      
      rankings.forEach(ranking => {
        ranking.forEach(({idx, rank}) => {
          const score = 1 / (k + rank);
          scores.set(idx, (scores.get(idx) || 0) + score);
        });
      });
      
      return scores;
    }

## 📚 涉及知识点

- [ ] **余弦相似度公式**
  ```
  similarity = dot(a, b) / (||a|| × ||b||)
  范围: [-1, 1]
  ```

- [ ] **RRF（Reciprocal Rank Fusion）算法**
  ```
  score = Σ(1 / (k + rank_i))
  k = 60（常数，避免前排差距过大）
  ```

- [ ] **TF-IDF 算法**
  ```
  TF = 词在文档中出现次数 / 文档总词数
  IDF = log(总文档数 / 包含该词的文档数)
  TF-IDF = TF × IDF
  ```

- [ ] **中文分词**（简单方案：按字拆分）

- [ ] **向量存储的两种模式**
  - FileStore: 本地 JSON，支持混合搜索
  - ChromaDB: 外部服务，纯向量搜索

## ⚠️ 注意事项

1. **相似度阈值的选择**：0.45 是经验值，需根据实际效果调整
2. **RRF 的 k 值**：k=60 是论文推荐值，可实验调整
3. **TF-IDF 性能**：大规模文档需考虑倒排索引优化
4. **向量维度**：智谱 embedding-3 返回的维度（确认是 1024 还是 2048）
5. **JSON 文件大小**：向量数据会随文档增多而增大，生产需迁移到数据库
