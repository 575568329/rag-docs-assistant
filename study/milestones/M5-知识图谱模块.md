import { z } from 'zod';

const EntitySchema = z.object({
  name: z.string(),
  type: z.enum(['Person', 'Organization', 'Location', 'Concept', 'Event', 'Document']),
  description: z.string().optional(),
});

const RelationSchema = z.object({
  source: z.string(),
  target: z.string(),
  type: z.enum(['BELONGS_TO', 'RELATED_TO', 'LOCATED_AT', 'PART_OF', 'CAUSED_BY']),
});

const ExtractionResultSchema = z.object({
  entities: z.array(EntitySchema),
  relations: z.array(RelationSchema),
});
```

  import { generateObject } from 'ai';
  import { createOpenAI } from '@ai-sdk/openai';
  
  export async function extractEntities(text: string, kbId: string, docId: string) {
    const zhipu = createOpenAI({
      baseURL: 'https://open.bigmodel.cn/api/paas/v4',
      apiKey: process.env.ZHIPU_API_KEY,
    });
    
    const { object } = await generateObject({
      model: zhipu('glm-4-flash'),
      schema: ExtractionResultSchema,
      prompt: `从以下文本中提取实体和关系：
  
  ${text}
  
  要求：
  1. 识别关键实体（人物、组织、地点、概念、事件、文档）
  2. 识别实体之间的关系
  3. 返回结构化数据`,
    });
    
    return {
      entities: object.entities.map(e => ({
        ...e,
        kbId,
        docId,
      })),
      relations: object.relations,
    };
  }
  ```

### 4. 实现图谱存储 (`src/lib/graph-store.ts`)

#### 4.1 初始化 Graphology
```typescript
import Graph from 'graphology';
import fs from 'fs';

const GRAPH_FILE = 'data/graph.json';

class GraphStore {
  private graph: Graph;
  
  constructor() {
    this.graph = new Graph();
    this.load();
  }
  
  private load() {
    if (fs.existsSync(GRAPH_FILE)) {
      const data = JSON.parse(fs.readFileSync(GRAPH_FILE, 'utf-8'));
      this.graph.import(data);
    }
  }
  
  private save() {
    fs.writeFileSync(GRAPH_FILE, JSON.stringify(this.graph.export()));
  }
}
```

#### 4.2 核心方法
- [ ] **addNode**
  ```typescript
  addNode(nodeId: string, label: string, type: EntityType, properties: Record<string, any>) {
    if (!this.graph.hasNode(nodeId)) {
      this.graph.addNode(nodeId, {
        label,
        type,
        ...properties,
      });
    }
    this.save();
  }
  ```

- [ ] **addEdge**
  ```typescript
  addEdge(source: string, target: string, relationType: RelationType, properties?: Record<string, any>) {
    if (this.graph.hasNode(source) && this.graph.hasNode(target)) {
      if (!this.graph.hasEdge(source, target)) {
        this.graph.addEdge(source, target, {
          relationType,
          ...properties,
        });
      }
    }
    this.save();
  }
  ```

- [ ] **queryGraph (按 kbId 过滤)**
  ```typescript
  queryGraph(kbId?: string): { nodes: any[], edges: any[] } {
    const nodes: any[] = [];
    const edges: any[] = [];
    
    this.graph.forEachNode((node, attrs) => {
      if (!kbId || attrs.kbId === kbId) {
        nodes.push({ id: node, ...attrs });
      }
    });
    
    this.graph.forEachEdge((edge, attrs, source, target) => {
      if (nodes.find(n => n.id === source) && nodes.find(n => n.id === target)) {
        edges.push({ id: edge, source, target, ...attrs });
      }
    });
    
    return { nodes, edges };
  }
  ```

- [ ] **searchNodes (关键词搜索)**
  ```typescript
  searchNodes(query: string, kbId?: string): string[] {
    const matches: string[] = [];
    
    this.graph.forEachNode((node, attrs) => {
      if ((!kbId || attrs.kbId === kbId) && attrs.label.includes(query)) {
        matches.push(node);
      }
    });
    
    return matches;
  }
  ```

- [ ] **getNeighborhood (获取邻域)**
  ```typescript
  getNeighborhood(nodeId: string, depth: number = 1): { nodes: any[], edges: any[] } {
    const visited = new Set<string>();
    const queue: Array<{node: string, level: number}> = [{node: nodeId, level: 0}];
    
    while (queue.length > 0) {
      const { node, level } = queue.shift()!;
      if (visited.has(node) || level > depth) continue;
      
      visited.add(node);
      
      if (level < depth) {
        this.graph.neighbors(node).forEach(neighbor => {
          queue.push({ node: neighbor, level: level + 1 });
        });
      }
    }
    
    // 返回访问到的节点和边
  }
  ```

### 5. 实现图谱查询 API (`src/app/api/graph/route.ts`)

- [ ] GET 方法
  ```typescript
  export async function GET(request: Request) {
    const { searchParams } = new URL(request.url);
    const kbId = searchParams.get('kbId') || undefined;
    const query = searchParams.get('query') || undefined;
    const nodeId = searchParams.get('nodeId') || undefined;
    
    const graphStore = new GraphStore();
    
    if (nodeId) {
      // 返回节点邻域
      const result = graphStore.getNeighborhood(nodeId, 1);
      return Response.json(result);
    } else if (query) {
      // 搜索节点
      const nodeIds = graphStore.searchNodes(query, kbId);
      const result = graphStore.queryGraph(kbId);
      // 过滤到匹配的节点及其邻域
      return Response.json(result);
    } else {
      // 返回全图
      const result = graphStore.queryGraph(kbId);
      return Response.json(result);
    }
  }
  ```

## ⚠️ 注意事项

1. **实体抽取成本**：LLM 调用次数多，考虑批量处理或缓存
2. **图谱规模**：节点过多时力导图性能下降，考虑分页或聚类
3. **实体去重**：同名实体可能被重复创建，需要归一化（Entity Linking）
4. **关系的方向性**：Graphology 支持有向图和无向图，根据业务选择
5. **节点 ID 设计**：使用 `${type}-${name}` 保证唯一性
